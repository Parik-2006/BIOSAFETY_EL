void setup() {
  pinMode(2, INPUT);         // PIR
  pinMode(4, INPUT_PULLUP);  // Button
  pinMode(12, OUTPUT);       // Green LED
  pinMode(8, OUTPUT);        // Buzzer
  pinMode(10, OUTPUT);       // Relay

  Serial.begin(9600);

  digitalWrite(10, HIGH); // Relay OFF (active LOW)
}

void loop() {
  int pir = digitalRead(2);
  int button = digitalRead(4);

  Serial.print("PIR: ");
  Serial.print(pir);
  Serial.print(" Button: ");
  Serial.println(button);

  if (button == LOW) {

    if (pir == LOW) {
      // SAFE
      digitalWrite(12, HIGH);   // LED ON
      digitalWrite(8, LOW);     // Buzzer OFF
      digitalWrite(10, LOW);    // Relay ON
      Serial.println("SAFE → UV ON");
    } 
    else {
      // MOTION DETECTED
      digitalWrite(12, LOW);
      digitalWrite(8, HIGH);    // Buzzer ON
      digitalWrite(10, HIGH);   // Relay OFF
      Serial.println("MOTION → UV OFF");
    }

  } else {
    // IDLE
    digitalWrite(12, LOW);
    digitalWrite(8, LOW);
    digitalWrite(10, HIGH);   // Relay OFF
    Serial.println("IDLE");
  }

  delay(300);
}
